133std::string BarPlotGenerator::formatTitle(const std::string& filename) {
134 // Supprimez l'extension du fichier
135 std::string base = filename.substr(0, filename.find_last_of("."));
136
137 // Ne gardez que les lettres et ajoutez un espace entre chaque mot
138 std::string cleaned;
139 bool lastWasLetter = false;
140 for (char c : base) {
141 if (std::isalpha(c)) {
142 if (lastWasLetter) {
143 cleaned += std::tolower(c);
144 } else {
145 cleaned += std::toupper(c);
146 lastWasLetter = true;
147 }
148 } else {
149 cleaned += ' ';
150 lastWasLetter = false;
151 }
152 }
153
154 // Enlevez les espaces éventuels en début ou en fin de chaîne
155 cleaned.erase(0, cleaned.find_first_not_of(' ')); // trim start
156 cleaned.erase(cleaned.find_last_not_of(' ') + 1); // trim end
157
158 return cleaned;
159}
